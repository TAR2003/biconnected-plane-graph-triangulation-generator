\section{Introduction}
\begin{frame}{Introduction}
	\textbf{The Problem:}
	\begin{itemize}
		\item Generate \textbf{all possible triangulations} of biconnected plane graphs
		\item Previous methods fail with complex structures due to multi-edge creation
		\item Important applications in graphics, engineering, and chip design
	\end{itemize}

	\vspace{0.5cm}
	\textbf{Our Contribution:}
	\begin{itemize}
		\item Novel algorithm with \textbf{constant time per triangulation}
		\item Solves the multi-edge problem using \textbf{safe root} concept
		\item Practical for real-world applications
	\end{itemize}
\end{frame}


\section{Our Approach}
\begin{frame}{Our Approach: Key Innovations}
	\textbf{Novel Tree Structure:}
	\begin{itemize}
		\item Instead of a single global genealogical tree, we use \textbf{local genealogical trees} for each face/cycle
		\item Each face maintains its own tree of triangulations with parent-child relationships
		\item Enables independent processing of faces while avoiding multi-edge issues
	\end{itemize}

	\vspace{0.3cm}
	\textbf{Safe Root Concept:}
	\begin{itemize}
		\item Introduces \textbf{safe root vertex selection} to prevent multi-edge creation
		\item Uses incremental chord checking to find optimal root vertex position
		\item Ensures all initial chords from root vertex are valid and non-conflicting
	\end{itemize}
\end{frame}

\begin{frame}{Our Approach: Recursive Tree Construction}
	\textbf{Local Tree Generation:}
	\begin{itemize}
		\item For each face $F_i$, build a genealogical tree $\mathcal{T}(F_i)$
		\item Initialize with safe root triangulation where all chords emanate from safe root vertex
		\item Generate children by flipping generating chords while maintaining parent-child relationships
	\end{itemize}

	\vspace{0.3cm}
	\textbf{GlobalChordSet Management:}
	\begin{itemize}
		\item Maintain global set of all chords used across all faces
		\item Before adding any chord, check if it already exists in GlobalChordSet
		\item Prevents multi-edge creation by rejecting conflicting chords
		\item Dynamic addition/removal during recursive exploration
	\end{itemize}
\end{frame}

\begin{frame}{Our Approach: Combinatorial Integration}
	\textbf{Recursive Face Processing:}
	\begin{itemize}
		\item Process faces sequentially from $F_1$ to $F_k$
		\item For each partial triangulation $T$, recursively triangulate remaining faces
		\item Combine local triangulations $T_i$ to form complete graph triangulation $T + T_i$
	\end{itemize}

	\vspace{0.3cm}
	\textbf{Efficient Output Generation:}
	\begin{itemize}
		\item Output complete triangulation only when all faces are processed ($i =$ face.length)
		\item For intermediate states, recursively continue to next face
		\item Ensures only valid complete triangulations are output
	\end{itemize}

	\vspace{0.3cm}
	\textbf{Constant-Time Operations:}
	\begin{itemize}
		\item Safe root finding in $O(1)$ time per face
		      \ chord operations and GlobalChordSet updates in constant time
		\item Overall $O(1)$ time per triangulation generation
	\end{itemize}
\end{frame}

\section{Algorithm}

\begin{frame}{Algorithm Structure Overview}
	\textbf{Our algorithm consists of 5 main procedures:}
	
	\begin{enumerate}
		\item \textbf{StartTriangulation}$(G, k)$
		\begin{itemize}
			\item Entry point - initializes the process
		\end{itemize}
		
		\vspace{0.2cm}
		\item \textbf{FindAllTriangulations}$(F, T, i)$
		\begin{itemize}
			\item Generates all triangulations for face $F_i$
			\item Uses safe root as starting point
		\end{itemize}
		
		\vspace{0.2cm}
		\item \textbf{FindAllChildTriangulations}$(T_i, T, i)$
		\begin{itemize}
			\item Explores genealogical tree via chord flipping
		\end{itemize}
		
		\vspace{0.2cm}
		\item \textbf{FindSafeRoot}$(F, i)$
		\begin{itemize}
			\item Finds safe root vertex and creates initial triangulation
		\end{itemize}
		
		\vspace{0.2cm}
		\item \textbf{OutputTriangulation}$(T, T_i, i)$
		\begin{itemize}
			\item Outputs complete triangulation or continues to next face
		\end{itemize}
	\end{enumerate}
\end{frame}

\begin{frame}[fragile]{Find All Child Triangulations}
	\scriptsize
	\textbf{Input:} $T_i$ = current triangulation, $T$ = partial solution, $i$ = face index
	
	\begin{algorithmic}[1]
		\Procedure{FindAllChildTriangulations}{$T_i, T, i$}
		\State \Call{OutputTriangulation}{$T, T_i, i$}
		\If{$T_i$ has no generating chords}
		\State \Return
		\EndIf
		\State Let $(v_{b},v_{b'})$ be the \textbf{leftmost blocking chord} of $T_i$
		\ForAll{vertices $v_j$ where $j \geq b$}
		\If{$(v_1, v_j)$ is a chord of $T_i$ \textbf{or} $(v_1, v_j) \notin$ \textit{GlobalChordSet}}
		\State Add chord $(v_1, v_j)$ to \textit{GlobalChordSet}
		\State \Call{FindAllChildTriangulations}{$T_i(v_1, v_j), T, i$}
		\State Remove chord $(v_1, v_j)$ from \textit{GlobalChordSet}
		\EndIf
		\EndFor
		\EndProcedure
	\end{algorithmic}
\end{frame}

\begin{frame}[fragile]{Find All Triangulations for a Face}
	\scriptsize
	\textbf{Input:} $F$ = faces array, $T$ = partial solution, $i$ = current face index
	
	\begin{algorithmic}[1]
		\Procedure{FindAllTriangulations}{$F, T, i$}
		\State $n \gets$ number of vertices in face $F_i$
		\State $T_{ir} \gets$ \Call{FindSafeRoot}{$F, i$} \Comment{Time: $O(1)$ per triangulation}
		\State Add all chords of $T_{ir}$ to \textit{GlobalChordSet} \Comment{Time: $O(n)$}
		\State \Call{OutputTriangulation}{$T, T_{ir}, i$}
		\State $T_i \gets T_{ir}$
		\For{$j = n-1$ \textbf{down to} $3$}
		\If{$(v_1, v_j) \notin$ \textit{GlobalChordSet}}
		\State Add chord $(v_1, v_j)$ to \textit{GlobalChordSet}
		\State \Call{FindAllChildTriangulations}{$T_i(v_1, v_j), T, i$}
		\State Remove chord $(v_1, v_j)$ from \textit{GlobalChordSet}
		\EndIf
		\EndFor
		\State Remove all chords of $T_{ir}$ from \textit{GlobalChordSet} \Comment{Time: $O(n)$}
		\EndProcedure
	\end{algorithmic}
\end{frame}

\begin{frame}[fragile]{Output Triangulation}
	\normalsize
	\textbf{Input:} $T$ = partial solution, $T_i$ = current face triangulation, $i$ = face index
	
	\begin{algorithmic}[1]
		\Procedure{OutputTriangulation}{$T, T_i, i$}
		\If{$i = $ \textit{totalFaces}}
		\State \textbf{Output} complete triangulation $(T + T_i)$
		\Else
		\State \Call{FindAllTriangulations}{$F, T + T_i, i + 1$}
		\EndIf
		\EndProcedure
	\end{algorithmic}
	
	\vspace{0.5cm}
	\textbf{Note:} This recursive structure combines triangulations from individual faces to generate complete graph triangulations.
\end{frame}


\begin{frame}[fragile]{Find Safe Root}
	\scriptsize
	\textbf{Input:} $F$ = faces array, $i$ = face index
	
	\begin{algorithmic}[1]
		\Procedure{FindSafeRoot}{$F, i$}
		\State $v_{root} \gets$ \Call{FindSafeRootVertex}{$F, i$}
		\State Create triangulation by adding chords from $v_{root}$ to all non-neighbors
		\State \Return root triangulation $T_{ir}$
		\EndProcedure
	\end{algorithmic}

	\vspace{0.3cm}
	\begin{algorithmic}[1]
		\Procedure{FindSafeRootVertex}{$F, i$}
		\State $n \gets$ number of vertices in face $F_i$
		\State $startIndex \gets n - 1$
		\State $endIndex \gets 0$
		\While{$startIndex > endIndex + 1$}
		\If{chord exists between $v_{startIndex}$ and $v_{endIndex}$ in \textit{GlobalChordSet}}
		\State $startIndex \gets startIndex - 1$
		\Else
		\State $endIndex \gets endIndex + 1$
		\EndIf
		\EndWhile
		\State \Return $v_{endIndex}$ as safe root vertex
		\EndProcedure
	\end{algorithmic}
\end{frame}
\begin{frame}[fragile]{Main Algorithm - Start Triangulation}
	\normalsize
	\textbf{Input:} $G$ = biconnected plane graph, $k$ = number of faces
	
	\begin{algorithmic}[1]
		\Procedure{StartTriangulation}{$G, k$}
		\State Label the faces as $F_1, F_2, \ldots, F_k$ arbitrarily
		\State Initialize \textit{GlobalChordSet} $\gets \emptyset$
		\State \Call{FindAllTriangulations}{$F, \emptyset, 1$}
		\EndProcedure
	\end{algorithmic}
	
	\vspace{0.5cm}
	\textbf{Note:} This is the entry point of the algorithm. It processes faces sequentially, building up complete triangulations through recursive exploration.
\end{frame}

\section{Proofs}

\subsection{Proof of Completeness}

\begin{frame}{Proof of Completeness: Pruning Strategy}
	\textbf{Pruning Operations:}
	\begin{itemize}
		\item Algorithm prunes triangulations where blocking chord is a duplicate chord
		\item Duplicate chord = chord that occurred in earlier faces
		\item Prune entire branch when duplication detected
	\end{itemize}

	\vspace{0.3cm}
	\textbf{Theorem (Completeness via Pruning):}
	\begin{block}{}
		All children of a pruned branch would have been invalid triangulations.
	\end{block}

	\vspace{0.3cm}
	\textbf{Key Insight:}
	\begin{itemize}
		\item Need to prove: blocking chord is never flipped in children
		\item If blocking chord persists in all children, and it's a duplicate
		\item Then all children are invalid, so pruning is safe
	\end{itemize}
\end{frame}

\begin{frame}{Proof of Completeness: Blocking Chord Analysis}
	\textbf{Setup:}
	\begin{itemize}
		\item Let chord $a$-$b$ be a blocking chord in triangulation tree node
		\item Two points form triangles with $a$-$b$ chord on both sides:
		      \begin{itemize}
			      \item $c$: point on opposite side of root vertex
			      \item $r$: point on root side
		      \end{itemize}
	\end{itemize}

	\vspace{0.3cm}
	\textbf{Three Cases Based on Position of $r$:}
	\begin{enumerate}
		\item $r$ is on the left of root vertex
		\item $r$ is on the right of root vertex
		\item $r$ is at the root
	\end{enumerate}
\end{frame}

\begin{frame}{Proof of Completeness: Case 1}
	\textbf{Case 1: $r$ is on the left of the root vertex}

	\begin{itemize}
		\item Flipping $a$-$b$ chord $\Rightarrow$ results in $r$-$c$ chord
		\item Before flipping: $b$-$r$ chord was also a blocking chord
		\item $b$-$r$ positioned to the left of $a$-$b$
		\item Therefore: $a$-$b$ could not be leftmost blocking chord
		\item After flip: $r$-$b$ chord still exists and remains left of $c$-$r$
		\item Consequence: new $r$-$c$ chord cannot be leftmost blocking chord
	\end{itemize}

	\vspace{0.3cm}
	\textbf{Conclusion:}
	\begin{block}{}
		Flipping $a$-$b$ blocking chord is NOT permitted by algorithm in this case.
	\end{block}
\end{frame}

\begin{frame}{Proof of Completeness: Cases 2 \& 3}
	\textbf{Case 2: $r$ is on the right of the root vertex}
	\begin{itemize}
		\item Flipping $a$-$b$ chord $\Rightarrow$ produces $r$-$c$ chord
		\item However: $a$-$r$ chord is more to the left than $r$-$c$ chord
		\item This flip is invalid according to the algorithm
	\end{itemize}

	\vspace{0.3cm}
	\textbf{Case 3: $r$ is at the root}
	\begin{itemize}
		\item Flipping $a$-$b$ chord $\Rightarrow$ creates root-$c$ chord
		\item This gives child triangulation's root vertex better visibility than parent
		\item Violates the algorithm's constraints
	\end{itemize}

	\vspace{0.3cm}
	\textbf{Final Conclusion:}
	\begin{block}{}
		In all three cases, flipping a blocking chord is impossible. If blocking chord is duplicate, all children are invalid. Pruning is safe and complete.
	\end{block}
\end{frame}

\subsection{Existence of Valid Root Triangulation}

\begin{frame}{Existence of Valid Root: Definitions}
	\textbf{Definition (Root Triangulation):}
	\begin{block}{}
		Root of the tree of triangulations, constructed by selecting a root vertex and adding chords from this root vertex to all non-neighboring vertices of the current face.
	\end{block}

	\vspace{0.3cm}
	\textbf{Definition (Safe Root Vertex):}
	\begin{block}{}
		A vertex from which adding chords to every other non-neighboring node will NOT create a multi-edge problem. None of the previous faces share a chord where one endpoint is the root vertex and the other is a vertex of the current face.
	\end{block}
\end{frame}

\begin{frame}{Existence of Valid Root: Main Theorem}
	\textbf{Theorem (Existence of Safe Root):}
	\begin{block}{}
		In any face of a biconnected planar graph, there always exist at least two safe vertices that can serve as root vertices for valid triangulations.
	\end{block}

	\vspace{0.3cm}
	\textbf{Proof Strategy:}
	\begin{itemize}
		\item Consider a face with $n$ vertices, numbered 1 to $n$
		\item Since planar graph, chords can only exist in outer plane
		\item Analyze two cases:
		      \begin{enumerate}
			      \item No chords from previous faces
			      \item At least one chord from previous faces
		      \end{enumerate}
	\end{itemize}
\end{frame}

\begin{frame}{Existence of Valid Root: Case 1}
	\textbf{Case 1: No chords from previous faces}

	\begin{itemize}
		\item If no chords from previous faces with both endpoints inside current face
		\item Then all $n$ vertices are eligible to be safe vertices
	\end{itemize}

	\vspace{0.5cm}
	\textbf{Simple but important base case!}
\end{frame}

\begin{frame}{Existence of Valid Root: Case 2 Setup}
	\textbf{Case 2: At least one chord from previous faces}

	\begin{itemize}
		\item Suppose chord $a$-$b$ on outer plane from previous face
		\item Both $a$ and $b$ are disqualified from being safe vertices
		\item Due to planarity: chords cannot cross each other
		\item Since $a$-$b$ chord exists, no chord on outer plane can overtake it
	\end{itemize}

	\vspace{0.3cm}
	\textbf{Consequence:}
	\begin{itemize}
		\item Nodes $(a+1, a+2, \ldots)$ might have connections between them
		\item They CANNOT have chords with nodes $(a-1, \ldots, b+1)$
		\item (Would violate planarity by crossing $a$-$b$ chord)
	\end{itemize}

	\vspace{0.3cm}
	\textbf{This creates $T_n$ structure:}
	\begin{itemize}
		\item $n$ = number of internal vertices between endpoints of constraining chord
	\end{itemize}
\end{frame}

\begin{frame}{$T_n$ Structures Analysis}
	\textbf{Current face contains two complementary $T_n$ structures:}
	\begin{enumerate}
		\item Structure 1: $a, a+1, \ldots, b$
		\item Structure 2: $a, a-1, \ldots, b$
	\end{enumerate}

	\vspace{0.3cm}
	\textbf{Internal vertices:} numbered $1, \ldots, n-3$

	\vspace{0.3cm}
	\textbf{Range:}
	\begin{itemize}
		\item Minimum: $T_1$ (since $T_0$ would mean $a$ and $b$ are neighbors)
		\item Maximum: $T_{n-3}$ (accounting for endpoints and complementary structure)
	\end{itemize}
\end{frame}

\begin{frame}{$T_n$ Structures: Base Cases}
	\textbf{Base Case: $T_1$}
	\begin{itemize}
		\item One internal vertex $c$ between $a$ and $b$
		\item Both $a$ and $b$ are neighbors of $c$
		\item Any chord from $c$ is impossible
		\item Therefore: $c$ is a safe vertex 
	\end{itemize}

	\vspace{0.3cm}
	\textbf{Base Case: $T_2$}
	\begin{itemize}
		\item Vertices $c$ and $d$ between $a$ and $b$
		\item If $c$ has a chord, it must be with $b$ (only non-neighbor)
		      \begin{itemize}
			      \item Creates $T_1$ structure containing $d$, making $d$ safe 
		      \end{itemize}
		\item If $d$ has a chord, it must be with $a$
		      \begin{itemize}
			      \item Creates $T_1$ structure containing $c$, making $c$ safe 
		      \end{itemize}
	\end{itemize}
\end{frame}

\begin{frame}{$T_n$ Structures: General Case Algorithm}
	\textbf{Finding a Safe Vertex in $T_n$ in $O(n)$ time:}

	\begin{enumerate}
		\item Consider vertex $a+1$
		\item $a+1$ cannot create chords with neighbors $a$ and $a+2$
		\item Possible chord partners: $\{a+3, \ldots, b\}$
		\item Check from $b$ downward using hashset ($O(1)$ per check)
	\end{enumerate}

	\vspace{0.3cm}
	\textbf{Three outcomes:}
	\begin{itemize}
		\item If chord $(a+1, b)$ exists: Structure becomes $T_{n-1}$ with vertices $a+2, \ldots, b-1$
		\item If chord $(a+1, b-k)$ exists for some $k > 0$: Structure becomes $T_{n-k-1}$
		\item Else: $a+1$ is a safe vertex (no chords with any candidate) 
	\end{itemize}
\end{frame}

\begin{frame}{$T_n$ Structures: Conclusion}
	\textbf{In each iteration:}
	\begin{itemize}
		\item If chord found: $n$ decreases, eventually reaching $T_1$ or $T_2$
		\item If no chord found: vertex is safe
	\end{itemize}

	\vspace{0.3cm}
	\textbf{Since:}
	\begin{itemize}
		\item Every check reduces problem size or finds safe vertex
		\item We've proven $T_1$ and $T_2$ contain safe vertices
	\end{itemize}

	\vspace{0.3cm}
	\textbf{Therefore:}
	\begin{block}{}
		Every $T_n$ structure contains at least one safe vertex.
	\end{block}

	\vspace{0.3cm}
	\textbf{Final Conclusion:}
	\begin{itemize}
		\item Face with no multi-edges: all $n$ vertices can be safe
		\item Face with at least one multi-edge: at least two complementary $T_n$ structures, each with at least one safe vertex
		\item \textbf{Every face has at least two safe vertices} 
	\end{itemize}
\end{frame}

\begin{frame}{Completeness Summary}
	\textbf{We have established:}
	\begin{enumerate}
		\item \textbf{Pruning preserves all valid triangulations}
		      \begin{itemize}
			      \item Blocking chords never flip in children
			      \item Duplicate blocking chords make all children invalid
			      \item Safe to prune entire branches
		      \end{itemize}

		      \vspace{0.2cm}
		\item \textbf{Safe root triangulations always exist}
		      \begin{itemize}
			      \item Every face has at least two safe vertices
			      \item Can always find valid starting triangulation
			      \item Independent of calculation order
		      \end{itemize}
	\end{enumerate}

	\vspace{0.3cm}
	\begin{block}{Conclusion}
		Our algorithm covers ALL valid planar triangulations of any planar biconnected graph.
	\end{block}
\end{frame}

\subsection{Time Complexity Analysis}

\begin{frame}{Time Complexity: Main Theorem}
	\textbf{Theorem (Amortized Linear Time Complexity):}
	\begin{block}{}
		The algorithm has amortized $O(1)$ time complexity per triangulation, or amortized linear time complexity overall.
	\end{block}

	\vspace{0.3cm}
	\textbf{Definition (Building Time):}
	\begin{itemize}
		\item Time required to calculate safe root and construct chords from root vertex for a single face
	\end{itemize}

	\vspace{0.3cm}
	\textbf{Components:}
	\begin{itemize}
		\item Finding a safe vertex: $O(n)$
		\item Adding all chords from the root: $O(n)$
		\item Total building time for face with $n$ vertices: $O(n)$
	\end{itemize}
\end{frame}

\begin{frame}{Time Complexity: Lower Bound on Triangulations}
	\textbf{Lemma (Lower Bound on Triangulation Count):}
	\begin{block}{}
		For a face with $n$ vertices, there are at least $O(n)$ valid triangulations.
	\end{block}

	\vspace{0.3cm}
	\textbf{Proof:}
	\begin{itemize}
		\item Every face has at least two safe vertices: $a$ and $c$
		\item Can reach $c$-root triangulation from $a$-root through chord flips
		\item Each flip produces a valid triangulation
	\end{itemize}

	\vspace{0.3cm}
	\textbf{Comparing $a$-root and $c$-root:}
	\begin{itemize}
		\item $a$-root: all chords have endpoint $a$
		\item $c$-root: all chords have endpoint $c$
		\item Only one chord may be shared: $a$-$c$
		\item All other $n-4$ chords must be different
		\item Requires $n-4$ flips $\Rightarrow$ at least $n-3$ triangulations
	\end{itemize}
\end{frame}

\begin{frame}{Time Complexity: Two-Face Graph}
	\textbf{Let $n_1$ and $n_2$ be vertices in faces 1 and 2:}

	\begin{align*}
		\text{Total building time: } B & = n_1 + n_1 \cdot n_2           \\
		\text{Total triangulations: } T & = n_1 \cdot n_2
	\end{align*}

	\vspace{0.3cm}
	\textbf{Since $n_2 \geq 3$:}
	\begin{align*}
		n_1 \cdot 1               & < n_1 \cdot n_2                   \\
		B = n_1 + n_1 \cdot n_2   & < n_1 \cdot n_2 + n_1 \cdot n_2   \\
		                          & = 2 \cdot n_1 \cdot n_2           \\
		                          & = 2T
	\end{align*}

	\vspace{0.3cm}
	\textbf{Total time complexity: $< 3T$}
\end{frame}

\begin{frame}{Time Complexity: Three-Face Graph}
	\textbf{Let $n_1$, $n_2$, $n_3$ be vertices in three faces:}

	\begin{align*}
		B & = n_1 + n_1 \cdot n_2 + n_1 \cdot n_2 \cdot n_3 \\
		T & = n_1 \cdot n_2 \cdot n_3
	\end{align*}

	\vspace{0.2cm}
	\textbf{From two-face case:} $n_1 + n_1 \cdot n_2 < 2 \cdot n_1 \cdot n_2$

	\vspace{0.2cm}
	\textbf{Therefore:}
	\begin{align*}
		B & < 2 \cdot n_1 \cdot n_2 + n_1 \cdot n_2 \cdot n_3
	\end{align*}

	\textbf{Since $n_3 \geq 3$, we have $2 < n_3$:}
	\begin{align*}
		2 \cdot n_1 \cdot n_2 & < n_1 \cdot n_2 \cdot n_3                                     \\
		B                      & < n_1 \cdot n_2 \cdot n_3 + n_1 \cdot n_2 \cdot n_3          \\
		                       & = 2T
	\end{align*}

	\textbf{Total time complexity: $< 3T$}
\end{frame}

\begin{frame}{Time Complexity: General Case}
	\textbf{For $k$ faces where each has at least 3 vertices:}

	\begin{align*}
		B & = n_1 + n_1 \cdot n_2 + \cdots + n_1 \cdot n_2 \cdots n_k \\
		T & = n_1 \cdot n_2 \cdots n_k
	\end{align*}

	\vspace{0.3cm}
	\textbf{Pattern holds by induction:}
	\begin{align*}
		n_1 + n_1 \cdot n_2 + \cdots + n_1 \cdot n_2 \cdots n_{k-1} & < 2 \cdot n_1 \cdot n_2 \cdots n_{k-1}
	\end{align*}

	\vspace{0.2cm}
	\textbf{Since $n_k \geq 3$:}
	\begin{align*}
		2 \cdot n_1 \cdot n_2 \cdots n_{k-1} & < n_1 \cdot n_2 \cdots n_k \\
		B                                     & < 2T
	\end{align*}

	\vspace{0.2cm}
	\textbf{Total time complexity: $< 3T$}

	\vspace{0.3cm}
	\begin{block}{Corollary}
		For any biconnected planar graph, algorithm takes total time $< 3 \times$ (number of triangulations). Therefore, amortized linear time complexity.
	\end{block}
\end{frame}

\subsection{Space Complexity Analysis}

\begin{frame}{Space Complexity}
	\textbf{Theorem (Linear Space Complexity):}
	\begin{block}{}
		The algorithm has $O(n)$ space complexity, where $n$ is the total number of vertices.
	\end{block}

	\vspace{0.3cm}
	\textbf{Proof:}

	When processing face $i$, must store information for faces $1, \ldots, i$:

	\begin{itemize}
		\item \textbf{Vertex information:} $n_1 + n_2 + \cdots + n_i$ vertices
		\item \textbf{Chord information:} For planar graph, $n-3$ chords total
		\item \textbf{Hashset for chord lookup:} Current chord information
	\end{itemize}

	\vspace{0.3cm}
	\textbf{Analysis:}
	\begin{itemize}
		\item Total vertices across all faces: $O(n)$
		\item Number of chords in planar graph: $O(n)$
		\item Hashset for current chord information: $O(n)$
	\end{itemize}

	\vspace{0.3cm}
	\textbf{Therefore: entire algorithm uses $O(n)$ space complexity.}
\end{frame}
